groups = df['group'].unique()
n_groups = len(groups)
fig, axes = plt.subplots(nrows=1, ncols=n_groups, figsize=(20, 5), sharey=True)

for i, group in enumerate(groups):
    # Seleccionamos los datos del grupo actual
    group_data = df[df['group'] == group]
    ax = axes[i]
    # Graficamos los datos de cada grupo en un scatter plot
    ax.scatter(group_data['x'], group_data['y'])
    ax.set_title(f'Group {group}')
    ax.set_xlabel('x')
    ax.set_ylabel('y')

plt.suptitle('Gráficas de dispersión de cada grupo')
plt.tight_layout()
plt.show()
